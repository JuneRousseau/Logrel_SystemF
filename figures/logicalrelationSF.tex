      \lrv{\xi}{\alpha} \eqdef&\ \xi(\alpha)\\
      %
      \lrv{\xi}{\unit} \eqdef&\ v = \unitt \\
      %
      \lrv{\xi}{\tau_{1} \times \tau_{2}}
      \eqdef&\
              \exists v_{1},\ v_{2}.\ v = (v_{1}, v_{2}) \wedge
              \lr{\xi}{\tau_{1}}{v_{1}} \wedge
              \lr{\xi}{\tau_{2}}{v_{2}} \\
      %
      \lrv{\xi}{\tau_{1} \rightarrow \tau_{2}}
      \eqdef&\
              \exists e.\ v = \lambda x.\ e \wedge
              ( \forall v'.\ \lr{\xi}{\tau_{1}}{v'} \Rightarrow
              \mathrm{Safe}_{\lrp{\xi}{\tau_{2}}}(e.[v'/x]))\\
      %
      \lrv{\xi}{\forall \alpha.\ \tau}
      \eqdef&\
              \exists e.\ v = \Lambda e \wedge
              ( \forall P.\ \mathrm{Safe}_{\lrp{((\alpha \mapsto P)::\xi)}{\tau}}(e))
  \end{flalign*}
