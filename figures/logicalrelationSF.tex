      \lrv{\sctx}{\alpha} \eqdef&\ \sctx(\alpha)\\

      \lrv{\sctx}{\typair{\ty_{1}}{\ty_{2}}}
      \eqdef&\
              \exists v_{1},~v_{2}.~v = \pair{v_{1}}{v_{2}} \wedge
              \lr{\sctx}{\ty_{1}}{v_{1}} \wedge
              \lr{\sctx}{\ty_{2}}{v_{2}} \\

      \lrv{\sctx}{\tyarrow{\ty_{1}}{\ty_{2}}}
      \eqdef&\
              \exists e.~v = \lam{x}{e} \wedge
              ( \forall v'.~\lr{\sctx}{\ty_{1}}{v'} \Rightarrow
              \safe_{\lrp{\sctx}{\ty_{2}}}(\subst{e}{v'}{x}))\\
      %
      \lrv{\sctx}{\tyforall{\alpha}{\ty}}
      \eqdef&\
              \exists e.~v = \tlam{e} \wedge
              ( \forall P.~\safe_{\lrp{((\alpha \mapsto P)::\sctx)}{\ty}}(e))
  \end{flalign*}
