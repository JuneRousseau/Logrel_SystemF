\newcommand{\lrp}[2]{\llbracket #2 \rrbracket_{#1}}
\newcommand{\lr}[3]{\llbracket #2 \rrbracket_{#1}(#3)}
\newcommand{\lrv}[2]{\lr{#1}{#2}{v}}
\begin{figure}[ht]
  \begin{mathpar}
    \lrv{\xi}{\alpha} \eqdef \xi(\alpha) \\

    \lrv{\xi}{\mathrm{unit}} \eqdef v = \mathrm{tt} \\

    \lrv{\xi}{\tau_{1} \times \tau_{2}} \eqdef
    \exists v_{1},\ v_{2}.\ v = (v_{1}, v_{2}) \wedge
    \lr{\xi}{\tau_{1}}{v_{1}} \wedge
    \lr{\xi}{\tau_{2}}{v_{2}} \\

    \lrv{\xi}{\tau_{1} \rightarrow \tau_{2}} \eqdef
    \exists e.\ v = \lambda \_.\ e \wedge
    ( \forall v'.\ \lr{\xi}{\tau_{1}}{v'} \Rightarrow
    \mathrm{Safe_{\lrp{\xi}{\tau_{2}}}(e.[v'/])})\\

    \lrv{\xi}{\forall \_.\ \tau} \eqdef
    \exists e.\ v = \Lambda e \wedge
    ( \forall P.\
    \mathrm{Safe_{\lrp{(P::\xi)}{\tau}}(e)})\\
  \end{mathpar}
  \caption{\label{fig:logrelSF} Logical relation }
\end{figure}
