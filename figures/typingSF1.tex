\begin{figure}[ht]
  $\alpha \in \mathrm{string}$

  $\tau ::= \mathrm{unit}\
  |\ \alpha\
  |\ \tau \times \tau\
  |\ \tau \rightarrow \tau\
  |\ \forall \tau
  $

  $\Gamma ::= \bullet\ |\ x:\tau,\Gamma$

  $\Delta ::= \bullet\ |\ \alpha,\Delta$

  \centering
  \begin{mathpar}
    \inferH
    {T-Unit}
    {  }
    {\typed{ \Delta; \Gamma}{\mathrm{tt}}{\mathrm{unit}}}
    \and

    \inferH
    {T-Var}
    {\Gamma (x) = \tau}
    {\typed{ \Delta; \Gamma}{x}{\tau}}
    \and

    \inferH
    {T-Prod}
    {\typed{ \Delta; \Gamma}{e_{1}}{\tau_{1}} \\ \typed{ \Delta; \Gamma}{e_{2}}{\tau_{2}}}
    {\typed{ \Delta; \Gamma}{(e_{1}, e_{2})}{\tau_{1} \times \tau_{2}}}
    \and

    \inferH
    {T-Fst}
    {\typed{ \Delta; \Gamma}{e}{\tau_{1} \times \tau_{2}}}
    {\typed{ \Delta; \Gamma}{\mathrm{fst}\ e}{\tau_{1}}}
    \and

    \inferH
    {T-Snd}
    {\typed{ \Delta; \Gamma}{e}{\tau_{1} \times \tau_{2}}}
    {\typed{ \Delta; \Gamma}{\mathrm{snd}\ e}{\tau_{2}}}
    \and

    \inferH
    {T-Abs}
    {\typed{ \Delta; x:\tau_{1}, \Gamma}{e}{\tau_{2}}}
    {\typed{ \Delta; \Gamma}{\lambda x. e}{\tau_{1} \rightarrow \tau_{2}}}
    \and

    \inferH
    {T-App}
    {\typed{ \Delta; \Gamma}{e_{1}}{\tau_{2} \rightarrow \tau_{1}}
      \\
      \typed{ \Delta; \Gamma}{e_{2}}{\tau_{2}}}
    {\typed{ \Delta; \Gamma}{e_{1}\ e_{2}}{\tau_{1}}}
    \and

    \inferH
    {T-TAbs}
    {\typed{\alpha, \Delta; \Gamma}{e}{\tau}
      \\
      \mathrm{\alpha is not free in \Gamma}
    {\typed{ \Delta; \Gamma}{\Lambda e}{\forall \alpha, \tau}}
    \and

    \inferH
    {T-TApp}
    {\typed{ \Delta; \Gamma}{\Lambda e}{\forall \alpha, \tau}}
    {\typed{ \Delta; \Gamma}{e\ \_}{\tau.[\tau'/\alpha]}}
    \and
  \end{mathpar}

\end{figure}
